Cria a tabela principal CREATE TABLE IF NOT EXISTS jogos_data ( id TEXT PRIMARY KEY, data JSONB NOT NULL DEFAULT '{}', updated_at TIMESTAMPTZ DEFAULT NOW() ); -- 2. Insere registros iniciais INSERT INTO jogos_data (id, data) VALUES ('master_schools', '[]') ON CONFLICT (id) DO NOTHING; -- 3. Permite acesso público (leitura e escrita) ALTER TABLE jogos_data ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "Allow all access" ON jogos_data; CREATE POLICY "Allow all access" ON jogos_data FOR ALL USING (true) WITH CHECK (true); -- 4. Função para atualizar timestamp automaticamente CREATE OR REPLACE FUNCTION update_timestamp() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql; DROP TRIGGER IF EXISTS update_jogos_data_timestamp ON jogos_data; CREATE TRIGGER update_jogos_data_timestamp BEFORE UPDATE ON jogos_data FOR EACH ROW EXECUTE FUNCTION update_timestamp(); -- 5. Habilitar Realtime (para sync em tempo real) ALTER PUBLICATION supabase_realtime ADD TABLE jogos_data;
